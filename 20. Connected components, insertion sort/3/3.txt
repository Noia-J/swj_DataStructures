 * 학번 이름
 * 본인은 이 소스파일을 다른 사람의 소스를 복사하지 않고 직접 작성하였습니다.
 * * 자료구조응용 과제 20 - 3번
 * 구조체 배열에 대한 삽입 정렬 (Insertion Sort)
 */

#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>
#include <stdlib.h>

// 레코드 구조체 정의
typedef struct {
    int key;
    char name[20];
    int grade;
} element;

// 함수 원형 선언
void insertionSort(element a[], int n);
void insert(element e, element a[], int i);

int main() {
    FILE *fp_in, *fp_out;
    int n;
    element *list;

    // 입력 파일 열기
    fp_in = fopen("input.txt", "r");
    if (fp_in == NULL) {
        fprintf(stderr, "Cannot open input.txt\n");
        exit(1);
    }

    // 데이터 개수 읽기
    fscanf(fp_in, "%d", &n);

    // 동적 메모리 할당 (1-based indexing을 위해 n+1 할당)
    list = (element*)malloc(sizeof(element) * (n + 1));
    if (list == NULL) {
        fprintf(stderr, "Memory allocation failed\n");
        exit(1);
    }

    // 데이터 읽기 (인덱스 1부터 저장)
    printf("<<<<<<<<<<<<<< Input List >>>>>>>>>>>>>\n");
    for (int i = 1; i <= n; i++) {
        fscanf(fp_in, "%d %s %d", &list[i].key, list[i].name, &list[i].grade);
        printf("< %d, %s, %d >\n", list[i].key, list[i].name, list[i].grade);
    }
    fclose(fp_in);

    // 삽입 정렬 수행
    insertionSort(list, n);

    // 정렬된 결과 출력 및 파일 저장
    fp_out = fopen("output.txt", "w");
    if (fp_out == NULL) {
        fprintf(stderr, "Cannot open output.txt\n");
        exit(1);
    }

    printf("\n<<<<<<<<<<<<<< Sorted List >>>>>>>>>>>>>\n");
    for (int i = 1; i <= n; i++) {
        printf("< %d, %s, %d >\n", list[i].key, list[i].name, list[i].grade);
        fprintf(fp_out, "< %d, %s, %d >\n", list[i].key, list[i].name, list[i].grade);
    }
    fclose(fp_out);

    free(list);
    return 0;
}

// 정렬된 리스트 a[1:i]에 원소 e를 삽입하는 함수 (Program 7.4)
void insert(element e, element a[], int i) {
    a[0] = e; // 보초값(sentinel) 설정
    while (e.key < a[i].key) {
        a[i + 1] = a[i];
        i--;
    }
    a[i + 1] = e;
}

// 삽입 정렬 함수 (Program 7.5)
void insertionSort(element a[], int n) {
    int j;
    for (j = 2; j <= n; j++) {
        element temp = a[j];
        insert(temp, a, j - 1);
    }
}
