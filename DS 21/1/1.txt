#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>
#include <stdlib.h>

#define MAX_SIZE 100
#define SWAP(x, y, t) { t = x; x = y; y = t; }

typedef struct {
    int key;
} element;

int callCount = 0;
int n; // 배열 전체 크기 (출력용)

// 현재 배열 상태와 left, right 인덱스를 출력하는 함수
void printState(element a[], int left, int right) {
    for (int i = 1; i <= n; i++) {
        printf("%4d", a[i].key); // 각 숫자를 4칸 폭으로 출력
    }
    printf(": %4d %4d\n", left, right); // : left right 출력
}

void quickSort(element a[], int left, int right) {
    int pivot, i, j;
    element temp;

    // 1. 함수 진입 시 무조건 상태 출력 및 카운트 증가 (이미지와 동일하게 맞춤)
    printState(a, left, right);
    callCount++;

    // 2. 기저 사례 체크 (출력 후에 체크해야 11 11 같은 leaf 호출도 찍힘)
    if (left < right) {
        i = left;
        j = right + 1;
        pivot = a[left].key;

        do {
            do { i++; } while (i <= right && a[i].key < pivot);
            do { j--; } while (j >= left && a[j].key > pivot);
            if (i < j) SWAP(a[i], a[j], temp);
        } while (i < j);

        SWAP(a[left], a[j], temp);

        quickSort(a, left, j - 1);
        quickSort(a, j + 1, right);
    }
}

int main() {
    FILE* fp_in = fopen("input.txt", "r");
    element a[MAX_SIZE];
    int i;

    if (fp_in == NULL) return 1;

    fscanf(fp_in, "%d", &n);
    for (i = 1; i <= n; i++) {
        fscanf(fp_in, "%d", &a[i].key);
    }

    // 입력 리스트 출력
    printf("<<<<<<<<<<<< Input List >>>>>>>>>>>>\n");
    for (i = 1; i <= n; i++) printf("%d ", a[i].key);
    printf("\n\n");

    printf("execution of quick sort ...\n");
    
    // 헤더 출력 [ 1] [ 2] ...
    for (i = 1; i <= n; i++) printf("[%2d] ", i);
    printf("[left right]\n");

    // 퀵 정렬 실행
    quickSort(a, 1, n);

    printf("calls of quick sort: %d\n", callCount);
    
    printf("\n<<<<<<<<<<<< Sorted List >>>>>>>>>>>>\n");
    for (i = 1; i <= n; i++) printf("%d ", a[i].key);
    printf("\n");

    fclose(fp_in);
    return 0;
}
