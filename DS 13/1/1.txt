#include <stdio.h>
#include <stdlib.h>
#include <string.h>

// MALLOC ИХХЉЗЮ: ЕПРћ ИоИ№ИЎ ЧвДч Йз ПРЗљ УГИЎ
#define MALLOC(p, s) \
	if (!((p) = malloc(s))) { \
		fprintf(stderr, "Insufficient memory"); \
		exit(EXIT_FAILURE); \
	}

// COMPARE ИХХЉЗЮ: ЕЮ СЄМі КёБГ
#define COMPARE(x, y) (((x) < (y)) ? -1 : ((x) == (y)) ? 0 : 1)

// --- БИЧі ММКЮ ЛчЧз Јч: ДйЧзНФ ГыЕх БИСЖУМ СЄРЧ ---
typedef struct polyNode *polyPointer;
//polyPointer ХИРд(Ся, struct polyNode БИСЖУМИІ АЁИЎХАДТ ЦїРЮХЭ)РЧ КЏМі newNodeИІ МБО№ЧЯАэ, getNode() ЧдМіИІ ШЃУтЧиМ­
ОђРК (ЛѕЗЮ ЧвДчЕЧАХГЊ РчЛчПыЕЧДТ) polyNode БИСЖУМРЧ ИоИ№ИЎ СжМвИІ Бз newNode КЏМіПЁ СяНУ ЧвДч(РњРх)ЧеДЯДй
typedef struct polyNode {
    int coef;
    int expon;
    polyPointer link;
} polyNode;

// --- БИЧі ММКЮ ЛчЧз Јч & Јщ: РќПЊ КЏМі МБО№ ---
polyPointer a = NULL, b = NULL, c = NULL; // ЧьДѕ ГыЕхИІ АЁСј ДйЧзНФ (ПјЧќ ПЌАс ИЎНКЦЎ)
polyPointer lastA = NULL, lastB = NULL;    // ДйЧзНФ aПЭ bПЁ ДыЧб last ЦїРЮХЭ
polyPointer avail = NULL;                 // ГыЕх ЧЎ (УМРЮ)

// --- ЧдМі ЧСЗЮХфХИРд МБО№ ---
polyPointer getNode(void);
void retNode(polyPointer node);
void erase(polyPointer *ptr);
void cerase(polyPointer *ptr);
void attach(int coefficient, int exponent, polyPointer *ptr);
void inputPolyCL(char *filename, polyPointer *p_header, polyPointer *p_last);
void printCList(polyPointer header);
polyPointer cpadd(polyPointer a, polyPointer b);

// --- ЧСЗЮБзЗЅ НУРлСЁ ---
int main(void) {
    char file_a[] = "a.txt";
    char file_b[] = "b.txt";

    // Јч inputPolyCL ЧдМі 2ШИ ШЃУтЧЯПЉ a, b Л§МК (ЧьДѕ ГыЕх ЦїЧд, last ЦїРЮХЭ РЏСі)
    inputPolyCL(file_a, &a, &lastA);
    inputPolyCL(file_b, &b, &lastB);

    // Јш printCList ШЃУтЧЯПЉ a, b СЄКИ УтЗТ
    printf("      a :");
    printCList(a);
    printf("      b :");
    printCList(b);

    // Јщ cpadd ШЃУтЧЯПЉ c = a + b АшЛъ
    c = cpadd(a, b);

    // Јъ printCList ШЃУтЧЯПЉ c СЄКИ УтЗТ
    printf("  a+b=c :");
    printCList(c);

    // Јы cerase 3Йј ШЃУтЧЯПЉ a, b, c ГыЕхИІ availПЁ ЙнГГ
    cerase(&a);
    cerase(&b);
    cerase(&c);

    // Јь erase ШЃУтЧЯПЉ avail ИЎНКЦЎРЧ И№Еч ГыЕх ИоИ№ИЎ ЧиСІ
    erase(&avail);

    return 0;
}

// --- ЧдМі СЄРЧ ---

/**
 * @brief ГыЕх ЧЎ(avail)ПЁМ­ ГыЕхИІ АЁСЎПРАХГЊ ЛѕЗЮ ЧвДч (Program 4.12 БтЙн)
 */
polyPointer getNode(void) {
    polyPointer node;
    if (avail) { // avail ИЎНКЦЎПЁ ГыЕхАЁ РжРИИщ
        node = avail;
        avail = avail->link;
    } else { // avail ИЎНКЦЎАЁ КёОюРжРИИщ ЛѕЗЮ ЧвДч
        MALLOC(node, sizeof(*node));
    }
    return node;
}

/**
 * @brief ЛчПыЧЯСі ОЪДТ ГыЕхИІ ГыЕх ЧЎ(avail)ПЁ ЙнШЏ (Program 4.13 БтЙн)
 */
void retNode(polyPointer node) {
    node->link = avail;
    avail = node;
}

/**
 * @brief УМРЮРИЗЮ БИЧіЕШ ИЎНКЦЎ(avail)РЧ И№Еч ГыЕх ИоИ№ИЎИІ ЧиСІ (Program 4.11 БтЙн)
 */
void erase(polyPointer *ptr) {
    polyPointer temp;
    while (*ptr) {
        temp = *ptr;
        *ptr = (*ptr)->link;
        free(temp); // mallocРИЗЮ ЧвДчЕШ ИоИ№ИЎ ЧиСІ
    }
}

/**
 * @brief ПјЧќ ПЌАс ИЎНКЦЎРЧ И№Еч ГыЕхИІ ГыЕх ЧЎ(avail)ПЁ ЙнШЏ (Program 4.14 БтЙн)
 * @param ptr ПјЧќ ИЎНКЦЎРЧ ЧьДѕ ГыЕхИІ АЁИЎХАДТ ЦїРЮХЭРЧ СжМв
 */
void cerase(polyPointer *ptr) {
    polyPointer temp;
    if (*ptr) { // ИЎНКЦЎАЁ NULLРЬ ОЦДЯИщ (ЧьДѕ ГыЕхАЁ СИРчЧЯИщ)
        temp = (*ptr)->link; // УЙ ЙјТА НЧСІ ГыЕх (ЖЧДТ ЧьДѕ РкНХ)
        (*ptr)->link = avail; // ИЖСіИЗ ГыЕхАЁ avail ИЎНКЦЎИІ АЁИЎХАЕЕЗЯ ПЌАс
        avail = temp;        // availРЧ НУРлРЛ ИЎНКЦЎРЧ НУРлРИЗЮ КЏАц
        *ptr = NULL;         // ИЎНКЦЎ ЦїРЮХЭИІ NULLЗЮ МГСЄ
    }
}

/**
 * @brief ДйЧзНФРЧ ИЖСіИЗПЁ Лѕ ЧзРЛ УпАЁ (Program 4.10 МіСЄ: getNode ЛчПы, int coef)
 * @param ptr ЧіРч ИЖСіИЗ ГыЕхИІ АЁИЎХАДТ ЦїРЮХЭРЧ СжМв (ЧдМі НЧЧр ШФ Лѕ ИЖСіИЗ ГыЕхИІ АЁИЎХД)
 */
void attach(int coefficient, int exponent, polyPointer *ptr) {
    polyPointer temp = getNode(); // MALLOC ДыНХ getNode ЛчПы
    temp->coef = coefficient;
    temp->expon = exponent;
    (*ptr)->link = temp; // ЧіРч ИЖСіИЗ ГыЕхАЁ Лѕ ГыЕхИІ АЁИЎХД
    *ptr = temp;         // ptrРЬ Лѕ ИЖСіИЗ ГыЕхИІ АЁИЎХАЕЕЗЯ ОїЕЅРЬЦЎ
}

/**
 * @brief ЦФРЯЗЮКЮХЭ ЕЅРЬХЭИІ РаОю ЧьДѕ ГыЕхИІ АЁСј ПјЧќ ПЌАс ИЎНКЦЎ ДйЧзНФ Л§МК
 * 'a' ЧќНФРК ОеПЁ УпАЁ, 'd' ЧќНФРК ЕкПЁ УпАЁЧЯИч last ЦїРЮХЭ АќИЎ
 */
void inputPolyCL(char *filename, polyPointer *p_header, polyPointer *p_last) {
    FILE *fp = fopen(filename, "r");
    if (!fp) {
        fprintf(stderr, "Cannot open file %s\n", filename);
        exit(EXIT_FAILURE);
    }

    char format;
    int coef, expon;

    // 1. ЧьДѕ ГыЕх Л§МК Йз УЪБтШ­
    *p_header = getNode();
    (*p_header)->expon = -1; // ЧьДѕ ГыЕхДТ СіМі -1ЗЮ БИКа
    (*p_header)->link = *p_header; // РкНХРЛ АЁИЎХАДТ ПјЧќ ИЎНКЦЎ
    *p_last = *p_header; // УЪБт ЛѓХТПЁМ­ДТ lastЕЕ ЧьДѕИІ АЁИЎХД

    fscanf(fp, " %c", &format); // 'a' ЖЧДТ 'd' ЧќНФ РаБт

    while (fscanf(fp, "%d %d", &coef, &expon) == 2) {
        polyPointer newNode = getNode();
        newNode->coef = coef;
        newNode->expon = expon;

        if (format == 'd') { // ГЛИВТїМј('d'): ИЎНКЦЎРЧ ГЁ(ЧьДѕ Ое)ПЁ УпАЁ
            newNode->link = *p_header; // Лѕ ГыЕхАЁ ЧьДѕИІ АЁИЎХД
            (*p_last)->link = newNode; // РЬРќ ИЖСіИЗ ГыЕхАЁ Лѕ ГыЕхИІ АЁИЎХД
            *p_last = newNode;         // last ЦїРЮХЭ ОїЕЅРЬЦЎ
        } else { // ПРИЇТїМј('a'): ИЎНКЦЎРЧ НУРл(ЧьДѕ ДйРН)ПЁ УпАЁ
            newNode->link = (*p_header)->link; // Лѕ ГыЕхАЁ БтСИ УЙ ГыЕхИІ АЁИЎХД
            (*p_header)->link = newNode;      // ЧьДѕАЁ Лѕ ГыЕхИІ АЁИЎХД
            // ИИОр РЬАЭРЬ УЙ ЙјТАЗЮ УпАЁЕШ ГыЕхЖѓИщ last ЦїРЮХЭЕЕ ОїЕЅРЬЦЎ
            if (*p_last == *p_header) {
                *p_last = newNode;
            }
        }
    }
    fclose(fp);
}

/**
 * @brief ЧьДѕ ГыЕхИІ АЁСј ПјЧќ ПЌАс ИЎНКЦЎ ДйЧзНФРЛ УтЗТ (НЧЧр ПЙНУ ЧќНФ)
 * @param header ДйЧзНФРЧ ЧьДѕ ГыЕх ЦїРЮХЭ
 */
void printCList(polyPointer header) {
    if (!header || header->link == header) { // ЧьДѕАЁ ОјАХГЊ ИЎНКЦЎАЁ КёОюРжРИИщ
        printf("\n");
        return;
    }
    polyPointer current = header->link; // УЙ ЙјТА НЧСІ ГыЕхКЮХЭ НУРл
    while (current != header) {        // ЧьДѕЗЮ ЕЙОЦПУ ЖЇБюСі ЙнКЙ
        printf(" %+d", current->coef);
        if (current->expon != 0) {
            printf("x^%d", current->expon);
        }
        current = current->link;
    }
    printf("\n");
}

/**
 * @brief ЧьДѕ ГыЕхИІ АЁСј ЕЮ ПјЧќ ПЌАс ИЎНКЦЎ ДйЧзНФ a, bИІ ДѕЧи cИІ ЙнШЏ (Program 4.15 БтЙн)
 */
polyPointer cpadd(polyPointer a, polyPointer b) {
    polyPointer startA, c, lastC;
    int sum;
    int done = 0; // FALSEПЁ ЧиДч

    startA = a;       // aРЧ НУРл(ЧьДѕ) БтЗЯ
    a = a->link;      // aИІ УЙ ЙјТА НЧСІ ГыЕхЗЮ РЬЕП
    b = b->link;      // bИІ УЙ ЙјТА НЧСІ ГыЕхЗЮ РЬЕП

    c = getNode();    // АсАњ ДйЧзНФ cРЧ ЧьДѕ ГыЕх Л§МК
    c->expon = -1;
    lastC = c;        // cРЧ ИЖСіИЗ ГыЕхИІ АЁИЎХГ ЦїРЮХЭ УЪБтШ­

    do {
        switch (COMPARE(a->expon, b->expon)) {
            case -1: // a->expon < b->expon
                attach(b->coef, b->expon, &lastC);
                b = b->link;
                break;
            case 0: // a->expon == b->expon
                // aАЁ ЧьДѕЗЮ ЕЙОЦПдДТСі ШЎРЮ (ПјЧќ ИЎНКЦЎ МјШИ ПЯЗс УМХЉ)
                if (startA == a) {
                    done = 1; // TRUE
                } else {
                    sum = a->coef + b->coef;
                    if (sum) attach(sum, a->expon, &lastC);
                    a = a->link;
                    b = b->link;
                }
                break;
            case 1: // a->expon > b->expon
                attach(a->coef, a->expon, &lastC);
                a = a->link;
                break;
        }
    } while (!done); // aАЁ Чб ЙйФћ ЕЙ ЖЇБюСі

    // bПЁ ГВРК ЧзЕщ УпАЁ (bДТ ОЦСї ЧьДѕПЁ ЕЕДоЧЯСі ОЪОвРЛ Мі РжРН)
    // ПјЧќ ИЎНКЦЎРЬЙЧЗЮ b != b->link АЁ ОЦДб, b->expon != -1 ЗЮ УМХЉЧиОп Чд
    while(b->expon != -1) {
         attach(b->coef, b->expon, &lastC);
         b = b->link;
    }

    lastC->link = c; // АсАњ ИЎНКЦЎ cИІ ПјЧќРИЗЮ ИИЕъ
    return c;
}
