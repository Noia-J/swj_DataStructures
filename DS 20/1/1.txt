/*
 * 학번 이름
 * 본인은 이 소스파일을 다른 사람의 소스를 복사하지 않고 직접 작성하였습니다.
 * * 자료구조응용 과제 20 - 1번
 * 무방향 그래프의 Connected Component 출력 프로그램
 */

#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>
#include <stdlib.h>

#define MAX_VERTICES 50
#define TRUE 1
#define FALSE 0

// 그래프 노드 구조체 정의
typedef struct node* nodePointer;
typedef struct node {
    int vertex;
    nodePointer link;
} node;

// 전역 변수
nodePointer graph[MAX_VERTICES]; // 인접 리스트 헤더 배열
int visited[MAX_VERTICES];       // 방문 여부 체크 배열
int n;                           // 정점의 수

// 함수 원형 선언
void insertEdge(int u, int v);
void printAdjList();
void dfs(int v);
void connected();

int main() {
    FILE* fp;
    int e; // 간선의 수
    int u, v;
    
    // 파일 열기
    fp = fopen("input.txt", "r");
    if (fp == NULL) {
        fprintf(stderr, "Cannot open input.txt\n");
        exit(1);
    }

    // 정점 수와 간선 수 읽기
    fscanf(fp, "%d %d", &n, &e);

    // 그래프 초기화
    for (int i = 0; i < n; i++) {
        graph[i] = NULL;
        visited[i] = FALSE;
    }

    // 간선 정보 읽기 및 인접 리스트 구성 (무방향 그래프이므로 양방향 추가)
    for (int i = 0; i < e; i++) {
        fscanf(fp, "%d %d", &u, &v);
        insertEdge(u, v);
        insertEdge(v, u);
    }

    fclose(fp);

    // 인접 리스트 출력
    printAdjList();

    // Connected Components 출력
    printf("\n<<<<<<<<<<<<< Connected Components >>>>>>>>>>\n");
    connected();

    return 0;
}

// 간선 추가 함수 (리스트의 맨 앞에 삽입)
void insertEdge(int u, int v) {
    nodePointer newNode = (nodePointer)malloc(sizeof(node));
    if (!newNode) {
        fprintf(stderr, "Memory allocation error\n");
        exit(1);
    }
    newNode->vertex = v;
    newNode->link = graph[u];
    graph[u] = newNode;
}

// 인접 리스트 출력 함수
void printAdjList() {
    printf("<<<<<<<<<<<<< Adjacency List >>>>>>>>>>>>>\n");
    for (int i = 0; i < n; i++) {
        printf("graph[%d] : ", i);
        for (nodePointer w = graph[i]; w; w = w->link) {
            printf("%5d", w->vertex);
        }
        printf("\n");
    }
}

// 깊이 우선 탐색 (DFS)
void dfs(int v) {
    nodePointer w;
    visited[v] = TRUE;
    printf("%5d", v); // 방문한 정점 출력

    for (w = graph[v]; w; w = w->link) {
        if (!visited[w->vertex]) {
            dfs(w->vertex);
        }
    }
}

// 연결 성분(Connected Components) 탐색 함수
void connected() {
    int i;
    int count = 1; // 컴포넌트 번호
    
    for (i = 0; i < n; i++) {
        if (!visited[i]) {
            printf("connected component %d : ", count++);
            dfs(i);
            printf("\n");
        }
    }
}
