#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>
#include <stdlib.h>

// 메모리 할당 매크로
#define MALLOC(p, s) \
	if (!((p) = malloc(s))) { \
		fprintf(stderr, "Insufficient memory"); \
		exit(EXIT_FAILURE); \
	}

#define MAX_VERTICES 50
#define FALSE 0
#define TRUE 1

// --- 그래프 노드 구조체 (인접 리스트용) ---
typedef struct node* nodePointer;
typedef struct node {
    int vertex;
    nodePointer link;
} node;

// --- 큐 노드 구조체 (BFS용) ---
typedef struct qNode* queuePointer;
typedef struct qNode {
    int vertex;
    queuePointer link;
} tQNode;

// --- 전역 변수 ---
nodePointer adjLists[MAX_VERTICES] = { NULL }; // 인접 리스트
short int visited[MAX_VERTICES]; // 방문 배열
int n; // 정점 수

queuePointer front = NULL, rear = NULL; // BFS용 단일 큐

// --- 함수 원형 선언 ---
void insertEdge(int u, int v);
void printAdjList(int n);
void bfs(int v);
void reset_visited(void);
void addq(int item);
int deleteq(void);


// --- main 함수 ---
int main(void) {
    FILE* fp;
    int e; // 간선의 수
    int u, v; // 간선을 구성하는 두 정점

    // 1. 파일 열기
    if ((fp = fopen("input.txt", "r")) == NULL) {
        fprintf(stderr, "Cannot open input.txt\n");
        exit(EXIT_FAILURE);
    }

    // 2. 정점 및 간선 수 읽기
    fscanf(fp, "%d %d", &n, &e);

    // 3. 인접 리스트 구성 (무방향 그래프)
    for (int i = 0; i < e; i++) {
        fscanf(fp, "%d %d", &u, &v);
        insertEdge(u, v);
        insertEdge(v, u);
    }
    fclose(fp);

    // 4. 인접 리스트 출력
    printf("<<<<<<<<<<<<< Adjacency List >>>>>>>>>>>\n");
    printAdjList(n);

    // 5. BFS 수행 및 출력
    printf("\n<<<<<<<<<<<<< Breadth First Search >>>>>>>>>>>\n");
    for (int i = 0; i < n; i++) {
        reset_visited(); // 방문 배열 초기화
        printf("bfs<%d>: ", i);
        bfs(i);
        printf("\n");
    }

    return 0;
}

/**
 * @brief 간선 (u, v)를 인접 리스트에 추가 (u -> v)
 */
void insertEdge(int u, int v) {
    nodePointer newNode;
    MALLOC(newNode, sizeof(*newNode));
    newNode->vertex = v;
    newNode->link = adjLists[u];
    adjLists[u] = newNode;
}

/**
 * @brief 인접 리스트 전체를 출력
 */
void printAdjList(int n) {
    for (int i = 0; i < n; i++) {
        printf("graph[%d]: ", i);
        nodePointer current = adjLists[i];
        while (current) {
            printf("%d ", current->vertex);
            current = current->link;
        }
        printf("\n");
    }
}

/**
 * @brief v를 시작으로 너비 우선 탐색(BFS) 수행 (큐 사용)
 * (Program 6.2)
 */
void bfs(int v) {
    nodePointer w;
    front = rear = NULL; // 1. 큐 초기화

    printf("%5d", v);
    visited[v] = TRUE; // 2. 시작 정점 방문 표시 및 출력
    addq(v);           // 3. 큐에 삽입

    while (front) { // 4. 큐가 비어있지 않은 동안
        v = deleteq(); // 5. 큐에서 정점을 하나 꺼냄
        
        // 6. 꺼낸 정점 v에 인접한 모든 w에 대해
        for (w = adjLists[v]; w; w = w->link) {
            if (!visited[w->vertex]) { // 7. 방문하지 않았다면
                printf("%5d", w->vertex);   // 8. 방문 정점 출력
                addq(w->vertex);            // 9. 큐에 삽입
                visited[w->vertex] = TRUE;  // 10. 방문 표시
            }
        }
    }
}

/**
 * @brief 전역 visited 배열을 모두 FALSE(0)로 초기화
 */
void reset_visited(void) {
    for (int i = 0; i < n; i++) {
        visited[i] = FALSE;
    }
}

/**
 * @brief 연결 큐의 뒤쪽에 아이템(정점) 추가
 * (Program 4.7 수정: 단일 큐 버전)
 */
void addq(int item) {
    queuePointer temp;
    MALLOC(temp, sizeof(*temp));
    temp->vertex = item;
    temp->link = NULL;
    
    if (front) { // 큐가 비어있지 않으면
        rear->link = temp;
    } else { // 큐가 비어있으면
        front = temp;
    }
    rear = temp; // 항상 rear는 새 노드를 가리킴
}

/**
 * @brief 연결 큐의 앞쪽에서 아이템(정점) 삭제 및 반환
 * (Program 4.8 수정: 단일 큐 버전)
 */
int deleteq(void) {
    queuePointer temp = front;
    int item;

    if (!temp) { // 큐가 비어있으면
        fprintf(stderr, "Queue is empty\n");
        return -1; // 오류
    }
    
    item = temp->vertex;
    front = temp->link; // front를 다음 노드로 이동
    
    if (!front) { // 만약 큐가 비게 되면
        rear = NULL;
    }

    free(temp);
    return item;
}
